#include <stdio.h>
#include <math.h>
int main() {
	int i, j;
	int num;
	scanf("%d", &num);
	if (3 <= num <= 19 && num % 2 == 1) {
		for (int i = 0; i < num; i++) {
			if (i <= round(num / 2)) {
				for (int j = 0; j <= i - 1; j++) {
					printf(" ");
				}
				for (int j = 1; j <= num - 2 * i; j++) {
					printf("*");
				}
				printf("\n");
			}

		}
		for (int i = num / 2 - 1; i >= 0; i--) {
			for (int j = 0; j < i; j++)
				printf(" ");
			for (int j = 0; j < num - 2 * i; j++)
				printf("*");
			printf("\n");
		}
	}
	else {
		printf("다시입력");
		
	}
	return 0;
	}
