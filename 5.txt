#include <stdio.h>

int reverse(int n) {
	int rev = 0;
	while (n != 0) {
		rev = rev * 10 + (n % 10);
		n /= 10;
	}
	return rev;
}

int is_even(int n) {
	if (n % 2 == 0) {
		return 1;
	}
	else {
		return 0;
	}
}

int main() {
	int num;
	printf("정수 입력: ");
	scanf("%d", &num);
	
	int reversed = reverse(num);
	printf("뒤집은 숫자: %d\n", reversed);

	if (is_even(reversed)) {
		printf("Even\n");
	}
	else {
		printf("Odd\n");
	}

	return 0;
}
