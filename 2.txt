#include <stdio.h>

int main() {
	int i, j;
	int num;
	scanf("%d", &num);
	for (int i = 1; i <= num; i++) {
		for (int j = 1; j <= num - i; j++) {
			printf(" #");
		}
		for (int j = 1; j <= i; j++) {
			if (j == i) {
				printf("  %d", j);
			}
			else {
				printf(" %d", j);
			}
			
		}
		printf("\n");
	}
	return 0;
}
